    let currentTab = "food";
    let order = [];

    function formatPrice(price) {
      return new Intl.NumberFormat('en-US', {
        minimumFractionDigits: 2,
        maximumFractionDigits: 2
      }).format(price);
    }

    function createSparkle() {
      const sparkles = ["✨", "💜", "⭐", "🌙"];
      const sparkle = document.createElement("div");
      sparkle.className = "sparkle";
      sparkle.textContent = sparkles[Math.floor(Math.random() * sparkles.length)];
      sparkle.style.left = Math.random() * window.innerWidth + "px";
      sparkle.style.top = Math.random() * window.innerHeight + "px";
      document.body.appendChild(sparkle);

      setTimeout(() => {
        sparkle.remove();
      }, 1200);
    }

    setInterval(createSparkle, 900);

    function switchTab(event, tab) {
      currentTab = tab;
      document.querySelectorAll(".menu-tab").forEach(btn => btn.classList.remove("active"));
      event.target.classList.add("active");
      displayMenu();
    }

    function displayMenu() {
      const menuItems = document.getElementById("menuItems");
      const items = menuData[currentTab];

      menuItems.innerHTML = items.map(item => `
        <div class="menu-item">
          <div class="menu-item-content">
            <div class="menu-item-name">
              ${item.name}
              ${item.badge ? `<span class="menu-item-badge">${item.badge}</span>` : ""}
            </div>
            <div class="menu-item-price">Price: $${formatPrice(item.price)}</div>
            <div class="menu-item-desc">${item.desc}</div>
          </div>
          <button class="add-btn" onclick="addToOrder('${item.id}', '${item.name.replace(/'/g, "\\'")}', ${item.price})">+ Add</button>
        </div>
      `).join("");
    }

    function addToOrder(id, name, price) {
      const existing = order.find(item => item.id === id);

      if (existing) {
        existing.qty++;
      } else {
        order.push({ id, name, price, qty: 1 });
      }

      updateOrder();
    }

    function increaseQty(idx) {
      order[idx].qty++;
      updateOrder();
    }

    function decreaseQty(idx) {
      if (order[idx].qty > 1) {
        order[idx].qty--;
      } else {
        order.splice(idx, 1);
      }
      updateOrder();
    }

    function removeFromOrder(idx) {
      order.splice(idx, 1);
      updateOrder();
    }

    function updateOrder() {
      const orderItems = document.getElementById("orderItems");

      if (order.length === 0) {
        orderItems.innerHTML = `<div class="empty-order">No items added yet</div>`;
      } else {
        orderItems.innerHTML = order.map((item, idx) => `
          <div class="order-item">
            <div class="order-item-info">
              <div class="order-item-name">${item.name}</div>
              <div class="order-item-price">$${formatPrice(item.price)} each</div>
            </div>
            <div class="qty-controls">
              <button class="qty-btn" onclick="decreaseQty(${idx})">−</button>
              <span class="qty-display">${item.qty}</span>
              <button class="qty-btn" onclick="increaseQty(${idx})">+</button>
            </div>
            <button class="order-item-remove" onclick="removeFromOrder(${idx})">✕</button>
          </div>
        `).join("");
      }

      const subtotal = order.reduce((sum, item) => sum + item.price * item.qty, 0);
      const total = subtotal;

      document.getElementById("subtotal").textContent = `$${formatPrice(subtotal)}`;
      document.getElementById("total").textContent = `$${formatPrice(total)}`;
    }

    function clearOrder() {
      order = [];
      updateOrder();
    }

    function submitOrder() {
      if (order.length === 0) {
        alert("Please add items to your order first!");
        return;
      }

      const customerName = document.getElementById("customerName").value.trim() || "Guest";
      const total = order.reduce((sum, item) => sum + item.price * item.qty, 0);

      let orderDetails = "";
      order.forEach(item => {
        orderDetails += `• ${item.name} x${item.qty} = $${formatPrice(item.price * item.qty)}\n`;
      });

      alert(
`Purple Haze Grill Order Submitted!

Customer: ${customerName}

${orderDetails}
Total: $${formatPrice(total)}

Where horsepower meets happy hour 💜`
      );

      clearOrder();
    }

    displayMenu();
